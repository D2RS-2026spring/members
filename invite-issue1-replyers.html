邀请 Issue 作者加入组织

使用 R 写一个简单的脚本，邀请 GitHub 组织（D2RS-2026spring）指定仓库（members）中的 Issue 1 的 Replyers 加入组织。

实施思路

gh R 包是 GitHub CLI 的 R 语言封装，它允许您直接在 R 中调用 gh CLI 的功能，并以更 R 友好的方式处理返回的数据（通常是 JSON ）。

前提条件

  1. Git：版本控制工具。

  2. R 和 RStudio：用于编写和运行 R 脚本。

  3. GitHub CLI (gh)：确保您已经安装并配置了 gh CLI，并且已经通过 gh auth login 进行了身份验证。

  4. gh R 包：您需要安装 gh R 包。

    install.packages("gh")

脚本实现

显示/隐藏代码 
# 加载必要的库
library(gh)
library(dplyr)
library(purrr) # 用于函数式编程，例如 map_chr

# --- 配置您的 GitHub 信息 ---
ORG_NAME <- "D2RS-2026spring"
REPO_NAME <- "members"
ISSUE_NUM = 1
# -----------------------------

# 1. 获取指定 Issue 的所有回复者（Replyers）
# gh 包会自动解析 JSON 响应为 R 列表或数据框
get_issue_replyers_gh <- function(org, repo, issue_number) {
  message(sprintf("正在使用 gh 包获取组织 %s 仓库 %s 的 Issue #%d 的回复者...", org, repo, issue_number))
  
  # 使用 gh::gh() 函数调用 GitHub API
  # /repos/:owner/:repo/issues/:issue_number/comments 是获取指定 Issue 评论的 API 路径
  # .send_headers = c(Accept = "application/vnd.github.v3+json") 确保获取 JSON 格式
  # per_page = 100 提高每页获取的数量，减少请求次数
  # .limit = Inf 表示获取所有页的数据
  comments <- gh::gh(
    "/repos/{owner}/{repo}/issues/{issue_number}/comments",
    owner = org,
    repo = repo,
    issue_number = issue_number,
    .send_headers = c(Accept = "application/vnd.github.v3+json"),
    per_page = 100,
    .limit = Inf
  )
  
  if (length(comments) == 0) {
    message("未找到任何回复。")
    return(character(0))
  }
  
  # 提取每条评论的作者登录名
  replyers <- comments %>%
    map_chr(~ .x$user$login) %>%
    unique()
  
  return(replyers)
}

# 2. 获取组织成员列表
get_organization_members_gh <- function(org) {
  message(sprintf("正在使用 gh 包获取组织 %s 的成员列表...", org))
  
  members <- gh::gh(
    "/orgs/{org}/members",
    org = org,
    .send_headers = c(Accept = "application/vnd.github.v3+json"),
    per_page = 100,
    .limit = Inf
  )
  
  if (length(members) == 0) {
    message("未找到任何组织成员。")
    return(character(0))
  }
  
  # 提取每个成员的登录名
  member_logins <- members %>%
    map_chr(~ .x$login) %>%
    unique()
  
  return(member_logins)
}

# 3. 获取组织中待处理的邀请列表
get_pending_invitations_gh <- function(org) {
  message(sprintf("正在使用 gh 包获取组织 %s 的待处理邀请列表...", org))
  
  invitations <- gh::gh(
    "/orgs/{org}/invitations",
    org = org,
    .send_headers = c(Accept = "application/vnd.github.v3+json"),
    per_page = 100,
    .limit = Inf
  )
  
  if (length(invitations) == 0) {
    message("当前没有待处理的邀请。")
    return(character(0))
  }
  
  # 提取每个待定受邀用户的登录名（login 可能为 NULL，需过滤）
  pending_logins <- invitations %>%
    map(~ .x$login) %>%
    keep(~ !is.null(.x)) %>%
    map_chr(~ .x) %>%
    unique()
  
  return(pending_logins)
}

# 4. 邀请用户加入组织
invite_user_to_org_gh <- function(org, username) {
  message(sprintf("正在邀请用户 %s 加入组织 %s...", username, org))

  tryCatch({
    # 1. 先获取用户的数字 ID
    user_info <- gh::gh(
      "/users/{username}",
      username = username,
      .send_headers = c(Accept = "application/vnd.github.v3+json")
    )
    user_id <- user_info$id

    gh::gh(
      "POST /orgs/{org}/invitations",
      org = org,
      invitee_id = user_id,
      role = "direct_member", # 设置角色为成员
      .send_headers = c(Accept = "application/vnd.github.v3+json")
    )
    message(sprintf("成功邀请 %s 加入组织 %s。", username, org))
  }, error = function(e) {
    stop(sprintf("未能邀请 %s 加入组织 %s。错误信息: %s", username, org, e$message))
  })
}

# --- 执行脚本 ---
# 让 R 的 gh 包使用 gh CLI 的 Token
Sys.setenv(GITHUB_PAT = system("gh auth token", intern = TRUE))


# 1. 获取 Issue 1 的所有回复者
issue_replyers <- get_issue_replyers_gh(ORG_NAME, REPO_NAME, ISSUE_NUM)

if (length(issue_replyers) == 0) {
  message("没有找到任何 Issue 回复者，或者获取失败。脚本结束。")
} else {
  message(sprintf("找到以下 Issue #%d 回复者: %s", ISSUE_NUM, paste(issue_replyers, collapse = ", ")))
  
  # 2. 获取组织成员
  org_members <- get_organization_members_gh(ORG_NAME)
  message(sprintf("当前组织成员: %s", paste(org_members, collapse = ", ")))
  
  # 3. 获取待处理的邀请列表
  pending_invitees <- get_pending_invitations_gh(ORG_NAME)
  if (length(pending_invitees) > 0) {
    message(sprintf("已有待处理邀请的用户: %s", paste(pending_invitees, collapse = ", ")))
  }
  
  # 4. 筛选出未在组织内且没有待处理邀请的回复者
  already_handled <- union(org_members, pending_invitees)
  replyers_to_invite <- setdiff(issue_replyers, already_handled)
  
  if (length(replyers_to_invite) == 0) {
    message("所有 Issue 回复者都已是组织成员或已有待处理邀请。无需邀请。")
  } else {
    message(sprintf("需要邀请以下回复者加入组织: %s", paste(replyers_to_invite, collapse = ", ")))
    
    # 5. 邀请这些回复者
    for (replyer in replyers_to_invite) {
      invite_user_to_org_gh(ORG_NAME, replyer)
      Sys.sleep(1) # 稍作延迟，避免请求频率过高
    }
    message("邀请过程完成。")
  }
}

message("脚本运行结束。")

# 从评论中提取学号和 GitHub 用户名的对应关系
extract_student_id <- function(comment) {
  body <- comment$body
  user <- comment$user$login
  
  # 尝试提取学号（13位数字，以2025开头）
  student_id <- NA_character_
  
  # 格式: "学号：2025303110116" 或 "学号: 2025303110116"
  student_id <- str_extract(body, "\\d{13}")
  
  data.frame(
    github_user = user,
    学号 = student_id,
    stringsAsFactors = FALSE
  )
}

# 批量提取映射关系
id_mapping <- map_dfr(issue1_comments, extract_student_id)
显示/隐藏代码 
# 去重：同一学号可能有多条评论，取第一条

- [ ] [**入学报到**](https://github.com/D2RS-2026spring/members/issues/1)：了解 GitHub 的基础用法

- [ ] [**我的主页**](https://github.com/D2RS-2026spring/members/issues/99)：尝试 AI 编程和项目管理
